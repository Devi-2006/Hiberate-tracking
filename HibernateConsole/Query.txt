

CREATE TABLE SENIOR_TBL (
    SENIOR_ID VARCHAR2(12) PRIMARY KEY,
    FULL_NAME VARCHAR2(150) NOT NULL,
    AGE NUMBER(3) CHECK (AGE >= 55),
    GENDER VARCHAR2(10),
    PRIMARY_PHONE VARCHAR2(20),
    CHRONIC_CONDITIONS VARCHAR2(300),
    EMERGENCY_CONTACT_NAME VARCHAR2(150),
    EMERGENCY_CONTACT_PHONE VARCHAR2(20),
    STATUS VARCHAR2(20) CHECK (STATUS IN ('ACTIVE','INACTIVE'))
);



CREATE TABLE PROGRAM_TBL (
    PROGRAM_ID VARCHAR2(10) PRIMARY KEY,
    PROGRAM_NAME VARCHAR2(150) NOT NULL,
    CATEGORY VARCHAR2(50),
    RECOMMENDED_DURATION_WEEKS NUMBER(3),
    SESSION_FREQUENCY VARCHAR2(50),
    STATUS VARCHAR2(20) CHECK (STATUS IN ('ACTIVE','INACTIVE','RETIRED'))
);

CREATE TABLE SESSION_TBL (
    SESSION_ID NUMBER(10) PRIMARY KEY,
    SENIOR_ID VARCHAR2(12),
    PROGRAM_ID VARCHAR2(10),
    SESSION_DATE DATE,
    TIME_SLOT VARCHAR2(20),
    CLINICIAN_NAME VARCHAR2(150),
    BLOOD_PRESSURE VARCHAR2(20),
    SUGAR_LEVEL NUMBER(5,1),
    SESSION_NOTES VARCHAR2(400),
    STATUS VARCHAR2(20) CHECK (STATUS IN ('SCHEDULED','COMPLETED','MISSED')),

    CONSTRAINT FK_SENIOR
        FOREIGN KEY (SENIOR_ID)
        REFERENCES SENIOR_TBL(SENIOR_ID),

    CONSTRAINT FK_PROGRAM
        FOREIGN KEY (PROGRAM_ID)
        REFERENCES PROGRAM_TBL(PROGRAM_ID)
);
create sequence sessionID_seq
START WITH 1000
MINVALUE 1000
MAXVALUE 9999
INCREMENT BY 1;
SELECT sessionID_seq.NEXTVAL from dual;
COMMIT;
DESC SESSION_TBL;
DESC SENIOR_TBL;
DESC PROGRAM_TBL;
